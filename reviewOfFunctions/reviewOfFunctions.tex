\begin{question}
  A number $x$, is to a function $f(x)$, is to a number $y$ as:
  \begin{solution}
    \begin{hint}
      If you are confused, see\link{http://en.wikipedia.org/wiki/Function_(mathematics)}.
    \end{hint}
    \begin{multiple-choice}
      \choice[correct]{Money, is to a pop machine, is to pop.}
      \choice{Money, is to pop, is to a pop machine.}
      \choice{Pop, is to a pop machine, is to money.}
      \choice{Pop, is to money, is to a pop machine.}
      \choice{A pop machine, is to pop, is to money.}
      \choice{A pop machine, is to money, is to pop.}
    \end{multiple-choice}
  \end{solution}
\end{question}

      \choice{For no real values of $x$}
    \end{multiple-choice}  
  \end{solution}

\begin{question}
  Which of the following statements is true?
  \begin{solution}
    \begin{hint}
      If you are confused, see\link{http://en.wikipedia.org/wiki/Inverse_function#Generalizations}.
    \end{hint}
    \begin{multiple-choice}
      \choice[correct]{$\sin(\sin^{-1}(10)) = 10$}
      \choice{$\sin^{-1}(\sin(10)) = 10$}
      \choice{$\sin^{-1}(x)$ is the inverse function of $\sin(x)$}
      \choice{$\sin^{-1}(x) = \frac{1}{\sin(x)}$}
    \end{multiple-choice}  
  \end{solution}
\end{question}



\begin{question}
  The expression $\log_b(x) = y$ is equivalent to:
  \begin{solution}
    \begin{hint}
      If you are confused, see\link{http://en.wikipedia.org/wiki/Logarithm}.
    \end{hint}
    \begin{multiple-choice}
      \choice[correct]{$b^y = x$}
      \choice{$b^x = y$}
      \choice{$x^y = b$}
      \choice{$y^x = b$}
      \choice{$x^b = y$}
      \choice{$y^b = x$}
    \end{multiple-choice}  
  \end{solution}
\end{question}
